/**
 * @event ControlMatrix#device:attached
 * @public
 */
/**
 * @event ControlMatrix#device:detaching
 * @public
 */
/**
 * @event ControlMatrix#device:detached
 * @public
 */

/**
 * @public
 * @class
 */
function DeviceRegistry (__ctx, __bus) {
  const __self = this

  const MODULE_ID = 'devices'

  /** @public */
  const getDeviceInfo = (deviceId, opts) => { }
  /** @public */
  const getDeviceList = (opts) => { }
  /** @public */
  const getDeviceLogs = (deviceId, opts) => { }

  /** @public */
  const toJSON = () => ({
    uid: MODULE_ID,
    attached: [],
  })

  /** @protected */
  const attach = (device, opts) => {
    __bus.emit('device:attaching', device.uid)
    __bus.emit('device:attached', device)
  }

  /** @protected */
  const detach = (deviceId, opts) => {
    __bus.emit('device:detaching', deviceId)
    __bus.emit('device:detached', deviceId)
  }

  /** @protected */
  const setup = () => {
    __bus.emit('subsystem:setup', MODULE_ID, __self)
  }

  /** @protected */
  const startup = () => {
    __bus.emit('subsystem:starting', MODULE_ID)
    __bus.emit('subsystem:available', MODULE_ID, __self)
  }

  /** @protected */
  const shutdown = () => {
    __bus.emit('subsystem:stopping', MODULE_ID, __self)
    __bus.emit('subsystem:unavailable', MODULE_ID)
  }

  // Interface
  this.uid = MODULE_ID
  this.setup = setup.bind(this)
  this.startup = startup.bind(this)
  this.shutdown = shutdown.bind(this)
  this.toJSON = toJSON.bind(this)
}

module.exports = DeviceRegistry
